SQL - COMMANDS
-- SELECT
SELECT * FROM website;
SELECT * FROM client;

-- SELECT specific columns
SELECT company_name, website_url FROM website;
SELECT client_name, contact FROM client;

-- INSERT
INSERT INTO website (company_name, website_url, client_name, website_type, domain, host)
VALUES ('ABC Company','abc.com','John','Ecommerce','abc.com','Hostinger');

INSERT INTO client (client_name, website_url, contact, domain, status)
VALUES ('John','abc.com','9876543210','abc.com','Active');

-- UPDATE
UPDATE website SET host='GoDaddy' WHERE company_name='ABC Company';
UPDATE client SET status='Inactive' WHERE client_name='John';

-- DELETE
DELETE FROM website WHERE company_name='ABC Company';
DELETE FROM client WHERE client_name='John';

-- WHERE
SELECT * FROM website WHERE domain='abc.com';
SELECT * FROM client WHERE status='Active';

-- DISTINCT
SELECT DISTINCT client_name FROM website;

-- ORDER BY
SELECT * FROM website ORDER BY company_name ASC;
SELECT * FROM client ORDER BY client_name DESC;

-- LIMIT
SELECT * FROM website LIMIT 5;
SELECT * FROM client LIMIT 10;

-- LIKE
SELECT * FROM website WHERE company_name LIKE '%ABC%';
SELECT * FROM client WHERE client_name LIKE '%John%';

-- BETWEEN
SELECT * FROM client WHERE contact BETWEEN 9000000000 AND 9999999999;

-- COUNT
SELECT COUNT(*) FROM website;
SELECT COUNT(*) FROM client;

-- GROUP BY
SELECT client_name, COUNT(*) FROM website GROUP BY client_name;

-- INNER JOIN
SELECT website.company_name, client.client_name, client.contact
FROM website
INNER JOIN client
ON website.client_name = client.client_name;

-- LEFT JOIN
SELECT website.company_name, client.contact
FROM website
LEFT JOIN client
ON website.client_name = client.client_name;

-- RIGHT JOIN
SELECT website.company_name, client.client_name
FROM website
RIGHT JOIN client
ON website.client_name = client.client_name;

-- ALIAS
SELECT w.company_name, c.contact
FROM website w
JOIN client c
ON w.client_name = c.client_name;

-- IN
SELECT * FROM website WHERE client_name IN ('John','David');

-- NOT IN
SELECT * FROM client WHERE status NOT IN ('Inactive');

-- IS NULL
SELECT * FROM website WHERE host IS NULL;

-- IS NOT NULL
SELECT * FROM website WHERE host IS NOT NULL;

-- HAVING
SELECT client_name, COUNT(*) AS total
FROM website
GROUP BY client_name
HAVING total > 1;

-- UNION
SELECT client_name FROM website
UNION
SELECT client_name FROM client;

-- UNION ALL
SELECT client_name FROM website
UNION ALL
SELECT client_name FROM client;
